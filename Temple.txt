
var util = import Utility

?foe = boss
	//util.UseMaskIfCan()
	?foe.distance <= 6
		equipL sword -big ice dI

		?foe.state = 112 & foe.time = 50
			util.DodgeIfCan()
		:?util.ShouldApplyPoison()
			equipR sword -big dP
		:?util.ShouldApplyFire()
			equipR sword -big fire dF	
		:
			equipR sword -big dL
	:?foe.distance <= 20
		util.DashIfCan()
	:
		util.Equips_NonCombat()
:
	?foe.distance <= 20
		?hp < maxhp
			equipL wand dL
		:
			equipL wand ice dI
		
		equipR shield ice A
	:
		util.Equips_NonCombat()
