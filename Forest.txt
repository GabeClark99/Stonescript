var util = import Utility

?foe = boss
	?foe = snail | foe = puff
		?foe.distance <= 6
			?foe.armor > 0
				equipL hammer poison D
			:
				equipL sword -big poison D

			?foe = puff & foe.distance < 5
				util.DodgeIfCan()
			:?hp < maxhp	
				equipR sword -big dL
			:
				equipR sword -big dP
		:?foe.distance <= 20
			equipL hammer poison D
			util.DashIfCan()
		:
			util.Equips_NonCombat()
	://shroom bosses
		//util.UseBladeIfCan()

		?foe.armor > 0 & foe.distance > 6 & foe.distance < 20
			equipL hammer poison D
			util.DashIfCan()

		?foe.distance <= 6
			util.UseMaskIfCan()
			?foe.armor > 0
				equipL hammer poison D
			:
				equipL sword poison D -big

			?util.ShouldApplyPoison()
				equipR sword poison dP -big
			:?hp < maxhp
				equipR sword dL -big
			:
				equipR sword poison dP -big
		:?foe.distance <= 20
			util.DashIfCan()
		:
			util.Equips_NonCombat()

: //adds
	?foe.distance <= 20
		equipL poison wand D

		?hp < maxhp
			equipR wand dL
		:
			equipR poison wand dP
	:
		util.Equips_NonCombat()
