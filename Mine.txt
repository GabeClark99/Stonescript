	?foe.state = 32 & foe.time > 20
		util.DodgeIfCan()

		util.UseMaskIfCan()

	:?foe.state = 33
		equipL sword -big dL

		?foe.distance > 6

		/*:?hp < maxhp
			equipR sword -big dL*/

			equipR sword -big aether +4
	:
		util.Equips_NonCombat()

: //not boss
	?foe.distance <= 6
		equipL sword big dL
		equipR shield aether A
		
		/*?foe.armor > 0
			equipL hammer aether
		:
			equipL big aether D

		?hp < maxhp
			equipR big dL
		:
			equipR big dP*/
	:?foe.distance <= 20
		util.DashIfCan()
	:
		util.Equips_NonCombat()
