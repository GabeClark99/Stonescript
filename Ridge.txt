var util = import Utility

?foe = boss | foe = wall
	?foe.distance <= 6
		util.UseMaskIfCan()

		equipL sword -big fire

		// TODO ice debuff
		?util.ShouldApplyFire()
			equipR sword -big fire dF
		:?hp < maxhp
			equipR sword -big dL
		:
			equipR sword -big dL
			
	:?foe.distance <= 20
		?foe.state = 133
			equipL wand vigor *8 +6

			// TODO ice debuff
			?hp < maxhp
				equipR wand dL
			:
				equipR wand vigor *9 +6
		:
			util.DashIfCan()
	:
		util.Equips_NonCombat()

: // non-boss
	?foe.distance <= 20
		?hp < maxhp
			equipL wand dL
		:
			equipL wand vigor *8 +6

		equipR shield fire
	:
		util.Equips_NonCombat()
